\usepackage[most]{tcolorbox}

\definecolor{solbg}[rgb]{HTML}{fcfcd0}

\newlength{\solmargins}
\setlength{\solmargins}{\dimexpr(\paperwidth-\textwidth)/2\relax}

\newtcolorbox{solbox2}[1][]{
    breakable,
    colback=solbg,
    colframe=solbg,
    spread sidewards,
    left*=0pt,
    right=\solmargins,
    #1,
}
    
\tcbset{list/.style={%
            enlarge left by=-\solmargins-#1,
            after skip=0pt,
            parbox=true,
        }%
}

\newenvironment{solution}{\begin{solbox2}}{\end{solbox2}}
